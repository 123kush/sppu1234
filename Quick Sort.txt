#include<iostream>
using namespace std;

void quick(int array[],int first,int last){
int p,f,l,temp,i;
f=first;
l=last;
p=first;
if(f<l){
    while(f<l){
        while(array[f]<=array[p]){
            f++;
        }
        while(array[l]>array[p]){
            l--;
        }
        if(f<l){
            temp=array[f];
            array[f]=array[l];
            array[l]=temp;
            }
    }
    temp=array[l];
    array[l]=array[p];
    array[p]=temp;
    quick(array,first,l-1);
    quick(array,l+1,last);

}
}
int main(){
    int array[100],n,i;
cout<<"Enter the size of the array: ";
cin>>n;
cout<<"Enter the"<<n<<"no of elements: ";
for( i=0;i<n;i++){
    cin>>array[i];
}
cout<<endl<<"Unsorted elements are: ";
for(i=0;i<n;i++)
    cout<<" "<<array[i];
quick(array,0,n-1);
cout<<endl<<endl<<"Sorted  elements usig quicksort are: ";
for(i=0;i<n;i++)
    cout<<" "<<array[i];
}
